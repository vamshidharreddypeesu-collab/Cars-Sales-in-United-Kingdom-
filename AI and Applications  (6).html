# Filter only hybrid cars
hybrid_cars = df[df["Fuel type"].str.lower() == "hybrid"]

# Group by Model and calculate average mileage
hybrid_mileage = hybrid_cars.groupby("Model")["Mileage"].mean().reset_index()

# Sort by mileage (optional: highest first)
hybrid_mileage = hybrid_mileage.sort_values(by="Mileage", ascending=False)

# Pie chart
plt.figure(figsize=(6,6))
plt.pie(
    hybrid_mileage["Mileage"], 
    labels=hybrid_mileage["Model"], 
    autopct="%1.1f%%", 
    startangle=200
)
plt.title("Mileage Share of Hybrid Car Models")
plt.show()
